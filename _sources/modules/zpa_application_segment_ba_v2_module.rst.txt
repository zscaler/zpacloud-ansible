.. Collection note

.. note::
    This module is part of the `zscaler.zpacloud collection <https://galaxy.ansible.com/ui/repo/published/zscaler/zpacloud/>`_ (version 2.0.3).

    It is not included in ``ansible-core``.
    To check whether it is installed, run :code:`ansible-galaxy collection list`.

    see :ref:`Requirements <ansible_collections.zscaler.zpacloud.zpa_application_segment_ba_v2_module_requirements>` for details.

.. Examples

Examples
--------

.. code-block:: yaml+jinja

    
    - name: Create an Application Segment Browser Access
      zscaler.zpacloud.zpa_application_segment_ba_v2:
        provider: "{{ zpa_cloud }}"
        name: Ansible_Application_Segment_BA
        description: Ansible_Application_Segment_BA
        enabled: true
        is_cname_enabled: true
        tcp_keep_alive: true
        passive_health_enabled: true
        select_connector_close_to_app: false
        health_check_type: "DEFAULT"
        health_reporting: "ON_ACCESS"
        bypass_type: "NEVER"
        icmp_access_type: false
        tcp_port_range:
          - from: "443"
            to: "443"
          - from: "4443"
            to: "4443"
        domain_names:
          - ba_access01.example.com
          - ba_access02.example.com
        segment_group_id: "216196257331368720"
        server_group_ids:
          - "216196257331368722"
        common_apps_dto:
          apps_config:
            - name: "ba_access01"
              description: "Description for common app"
              enabled: true
              domain: ba_access01.example.com
              application_port: "443"
              application_protocol: "HTTPS"
              certificate_id: 72058304855021564
            - name: "ba_access02"
              description: "Description for common app"
              enabled: true
              domain: ba_access02.example.com
              application_port: "4443"
              application_protocol: "HTTPS"
              certificate_id: "72058304855021564"
